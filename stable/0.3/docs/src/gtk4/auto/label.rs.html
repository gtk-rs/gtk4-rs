    /// not have enough room to display the entire string.

    /// [`pango::EllipsizeMode::None`][crate::pango::EllipsizeMode::None] has the side-effect that the label requests

    /// means that ellipsizing labels do not work well in notebook tabs, unless
    /// the `property::NotebookPage::tab-expand` child property is set to [`true`].

    pub fn ellipsize(mut self, ellipsize: pango::EllipsizeMode) -> Self {

    /// If the string contains underlines acting as mnemonics, you will have to
    /// set the `property::Label::use-underline` property to [`true`] in order
    /// for the label to display them.
    pub fn label(mut self, label: &str) -> Self {

        self
