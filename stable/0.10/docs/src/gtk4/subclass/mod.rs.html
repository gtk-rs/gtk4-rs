7/// Struct to hold a pointer and free it on `Drop::drop`
8pub(crate) struct PtrHolder<T, F: Fn(*mut T) + 'static>(*mut T, F);
9
10impl<T, F: Fn(*mut T) + 'static> Drop for PtrHolder<T, F> {
11    #[inline]
12    fn drop(&mut self) {
13        (self.1)(self.0)
14    }
15}
