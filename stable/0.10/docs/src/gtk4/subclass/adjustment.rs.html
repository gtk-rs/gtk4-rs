4//! Traits intended for subclassing [`Adjustment`].
5
6use glib::translate::*;
7
8use crate::{ffi, prelude::*, subclass::prelude::*, Adjustment};
9
10pub trait AdjustmentImpl: ObjectImpl + ObjectSubclass<Type: IsA<Adjustment>> {
11    fn changed(&self) {
12        self.parent_changed()
13    }
14
15    fn value_changed(&self) {
16        self.parent_value_changed()
17    }
18}
19
20pub trait AdjustmentImplExt: AdjustmentImpl {
21    fn parent_changed(&self) {
22        unsafe {
23            let data = Self::type_data();
24            let parent_class = data.as_ref().parent_class() as *mut ffi::GtkAdjustmentClass;
25            if let Some(f) = (*parent_class).changed {
26                f(self.obj().unsafe_cast_ref::<Adjustment>().to_glib_none().0)
27            }
28        }
29    }
30
31    fn parent_value_changed(&self) {
32        unsafe {
33            let data = Self::type_data();
34            let parent_class = data.as_ref().parent_class() as *mut ffi::GtkAdjustmentClass;
35            if let Some(f) = (*parent_class).value_changed {
36                f(self.obj().unsafe_cast_ref::<Adjustment>().to_glib_none().0)
37            }
38        }
39    }
40}
41
42impl<T: AdjustmentImpl> AdjustmentImplExt for T {}
43
44unsafe impl<T: AdjustmentImpl> IsSubclassable<T> for Adjustment {
45    fn class_init(class: &mut glib::Class<Self>) {
46        Self::parent_class_init::<T>(class);
47
48        assert_initialized_main_thread!();
49
50        let klass = class.as_mut();
51        klass.changed = Some(adjustment_changed::<T>);
52        klass.value_changed = Some(adjustment_value_changed::<T>);
53    }
54}
55
56unsafe extern "C" fn adjustment_changed<T: AdjustmentImpl>(ptr: *mut ffi::GtkAdjustment) {
57    let instance = &*(ptr as *mut T::Instance);
58    let imp = instance.imp();
59
60    imp.changed()
61}
62
63unsafe extern "C" fn adjustment_value_changed<T: AdjustmentImpl>(ptr: *mut ffi::GtkAdjustment) {
64    let instance = &*(ptr as *mut T::Instance);
65    let imp = instance.imp();
66
67    imp.value_changed()
68}
