4//! Traits intended for subclassing [`Button`](crate::Button).
5
6use glib::translate::*;
7
8use crate::{ffi, prelude::*, subclass::prelude::*, Button};
9
10pub trait ButtonImpl: ButtonImplExt + WidgetImpl {
11    /// Signal that causes the button to animate press then
12    ///    release. Applications should never connect to this signal, but use
13    ///    the @clicked signal.
14    fn activate(&self) {
15        self.parent_activate()
16    }
17
18    /// Signal emitted when the button has been activated (pressed and released).
19    fn clicked(&self) {
20        self.parent_clicked()
21    }
22}
23
24mod sealed {
25    pub trait Sealed {}
26    impl<T: super::ButtonImplExt> Sealed for T {}
27}
28
29pub trait ButtonImplExt: sealed::Sealed + ObjectSubclass {
30    fn parent_activate(&self) {
31        unsafe {
32            let data = Self::type_data();
33            let parent_class = data.as_ref().parent_class() as *mut ffi::GtkButtonClass;
34            if let Some(f) = (*parent_class).activate {
35                f(self.obj().unsafe_cast_ref::<Button>().to_glib_none().0)
36            }
37        }
38    }
39
40    fn parent_clicked(&self) {
41        unsafe {
42            let data = Self::type_data();
43            let parent_class = data.as_ref().parent_class() as *mut ffi::GtkButtonClass;
44            if let Some(f) = (*parent_class).clicked {
45                f(self.obj().unsafe_cast_ref::<Button>().to_glib_none().0)
46            }
47        }
48    }
49}
50impl<T: ButtonImpl> ButtonImplExt for T {}
51
52unsafe impl<T: ButtonImpl> IsSubclassable<T> for Button {
53    fn class_init(class: &mut glib::Class<Self>) {
54        Self::parent_class_init::<T>(class);
55
56        let klass = class.as_mut();
57        klass.activate = Some(button_activate::<T>);
58        klass.clicked = Some(button_clicked::<T>);
59    }
60}
61
62unsafe extern "C" fn button_activate<T: ButtonImpl>(ptr: *mut ffi::GtkButton) {
63    let instance = &*(ptr as *mut T::Instance);
64    let imp = instance.imp();
65
66    imp.activate()
67}
68
69unsafe extern "C" fn button_clicked<T: ButtonImpl>(ptr: *mut ffi::GtkButton) {
70    let instance = &*(ptr as *mut T::Instance);
71    let imp = instance.imp();
72
73    imp.clicked()
74}
