4//! Traits intended for subclassing [`ListBoxRow`].
5
6use glib::translate::*;
7
8use crate::{ffi, prelude::*, subclass::prelude::*, Actionable, ListBoxRow};
9
10pub trait ListBoxRowImpl:
11    WidgetImpl + ObjectSubclass<Type: IsA<ListBoxRow> + IsA<Actionable>>
12{
13    fn activate(&self) {
14        self.parent_activate()
15    }
16}
17
18pub trait ListBoxRowImplExt: ListBoxRowImpl {
19    fn parent_activate(&self) {
20        unsafe {
21            let data = Self::type_data();
22            let parent_class = data.as_ref().parent_class() as *mut ffi::GtkListBoxRowClass;
23            if let Some(f) = (*parent_class).activate {
24                f(self.obj().unsafe_cast_ref::<ListBoxRow>().to_glib_none().0)
25            }
26        }
27    }
28}
29
30impl<T: ListBoxRowImpl> ListBoxRowImplExt for T {}
31
32unsafe impl<T: ListBoxRowImpl> IsSubclassable<T> for ListBoxRow {
33    fn class_init(class: &mut glib::Class<Self>) {
34        Self::parent_class_init::<T>(class);
35
36        let klass = class.as_mut();
37        klass.activate = Some(row_activate::<T>);
38    }
39}
40
41unsafe extern "C" fn row_activate<T: ListBoxRowImpl>(ptr: *mut ffi::GtkListBoxRow) {
42    let instance = &*(ptr as *mut T::Instance);
43    let imp = instance.imp();
44
45    imp.activate()
46}
