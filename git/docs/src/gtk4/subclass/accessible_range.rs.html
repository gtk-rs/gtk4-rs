4//! Traits intended for implementing the [`AccessibleRange`] interface.
5
6use glib::translate::*;
7
8use crate::{ffi, prelude::*, subclass::prelude::*, AccessibleRange};
9
10pub trait AccessibleRangeImpl: AccessibleImpl + ObjectSubclass<Type: IsA<AccessibleRange>> {
11    fn set_current_value(&self, value: f64) -> bool {
12        self.parent_set_current_value(value)
13    }
14}
15
16pub trait AccessibleRangeImplExt: AccessibleRangeImpl {
17    // Returns true if the operation was performed, false otherwise
18    fn parent_set_current_value(&self, value: f64) -> bool {
19        unsafe {
20            let type_data = Self::type_data();
21            let parent_iface = type_data.as_ref().parent_interface::<AccessibleRange>()
22                as *const ffi::GtkAccessibleRangeInterface;
23
24            let func = (*parent_iface)
25                .set_current_value
26                .expect("no parent \"set_current_value\" implementation");
27
28            from_glib(func(
29                self.obj()
30                    .unsafe_cast_ref::<AccessibleRange>()
31                    .to_glib_none()
32                    .0,
33                value,
34            ))
35        }
36    }
37}
38
39impl<T: AccessibleRangeImpl> AccessibleRangeImplExt for T {}
40
41unsafe impl<T: AccessibleRangeImpl> IsImplementable<T> for AccessibleRange {
42    fn interface_init(iface: &mut glib::Interface<Self>) {
43        let iface = iface.as_mut();
44
45        iface.set_current_value = Some(accessible_range_set_current_value::<T>);
46    }
47}
48
49unsafe extern "C" fn accessible_range_set_current_value<T: AccessibleRangeImpl>(
50    accessible_range: *mut ffi::GtkAccessibleRange,
51    value: f64,
52) -> glib::ffi::gboolean {
53    let instance = &*(accessible_range as *mut T::Instance);
54    let imp = instance.imp();
55
56    imp.set_current_value(value).into_glib()
57}
