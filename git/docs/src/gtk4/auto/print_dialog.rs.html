619            }
620        }
621        unsafe {
622            let f: Box_<F> = Box_::new(f);
623            connect_raw(
624                self.as_ptr() as *mut _,
625                c"notify::title".as_ptr() as *const _,
626                Some(std::mem::transmute::<*const (), unsafe extern "C" fn()>(
627                    notify_title_trampoline::<F> as *const (),
628                )),
629                Box_::into_raw(f),
630            )
631        }
632    }
633}
