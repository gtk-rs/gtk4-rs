9impl From<cmp::Ordering> for Ordering {

58/// Controls how a widget deals with extra space in a single dimension.
59///
60/// Alignment only matters if the widget receives a “too large” allocation,
61/// for example if you packed the widget with the [`hexpand`][struct@crate::Widget#hexpand]
62/// property inside a [`Box`][crate::Box], then the widget might get extra space.
63/// If you have for example a 16x16 icon inside a 32x32 space, the icon
64/// could be scaled and stretched, it could be centered, or it could be
65/// positioned to one side of the space.
66///
67/// Note that in horizontal context `GTK_ALIGN_START` and `GTK_ALIGN_END`
68/// are interpreted relative to text direction.
69///
70/// Baseline support is optional for containers and widgets, and is only available
71/// for vertical alignment. `GTK_ALIGN_BASELINE_CENTER` and `GTK_ALIGN_BASELINE_FILL`
72/// are treated similar to `GTK_ALIGN_CENTER` and `GTK_ALIGN_FILL`, except that it
73/// positions the widget to line up the baselines, where that is supported.
74#[derive(Debug, Eq, PartialEq, Ord, PartialOrd, Hash, Clone, Copy)]
75#[non_exhaustive]
76#[doc(alias = "GtkAlign")]
77pub enum Align {
78    /// stretch to fill all space if possible, center if
79    ///   no meaningful way to stretch
80    #[doc(alias = "GTK_ALIGN_FILL")]
81    Fill,
82    /// snap to left or top side, leaving space on right or bottom
83    #[doc(alias = "GTK_ALIGN_START")]
84    Start,
85    /// snap to right or bottom side, leaving space on left or top
86    #[doc(alias = "GTK_ALIGN_END")]
87    End,
88    /// center natural width of widget inside the allocation
89    #[doc(alias = "GTK_ALIGN_CENTER")]
90    Center,
91    /// align the widget according to the baseline.
92    /// Use `GTK_ALIGN_BASELINE_FILL` instead
93    #[doc(alias = "GTK_ALIGN_BASELINE")]
94    Baseline,
95    /// a different name for `GTK_ALIGN_BASELINE`.
96    #[cfg(feature = "v4_12")]
97    #[cfg_attr(docsrs, doc(cfg(feature = "v4_12")))]
98    #[doc(alias = "GTK_ALIGN_BASELINE_FILL")]
99    BaselineFill,
100    /// stretch to fill all space, but align the baseline.
101    #[cfg(feature = "v4_12")]
102    #[cfg_attr(docsrs, doc(cfg(feature = "v4_12")))]
103    #[doc(alias = "GTK_ALIGN_BASELINE_CENTER")]
104    BaselineCenter,
105    #[doc(hidden)]
106    __Unknown(i32),
107}

109impl std::fmt::Display for Align {
110    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
111        write!(
112            f,
113            "Align::{}",
114            match *self {
115                Self::Fill => "Fill",
116                Self::Start => "Start",
117                Self::End => "End",
118                Self::Center => "Center",
119                Self::Baseline => "Baseline",
120                #[cfg(feature = "v4_12")]
121                Self::BaselineFill => "BaselineFill",
122                #[cfg(feature = "v4_12")]
123                Self::BaselineCenter => "BaselineCenter",
124                _ => "Unknown",
125            }
126        )
127    }
128}

130#[doc(hidden)]
131impl IntoGlib for Align {
132    type GlibType = ffi::GtkAlign;
133
134    #[inline]
135    fn into_glib(self) -> ffi::GtkAlign {
136        match self {
137            Self::Fill => ffi::GTK_ALIGN_FILL,
138            Self::Start => ffi::GTK_ALIGN_START,
139            Self::End => ffi::GTK_ALIGN_END,
140            Self::Center => ffi::GTK_ALIGN_CENTER,
141            #[cfg(not(feature = "v4_12"))]
142            Self::Baseline => ffi::GTK_ALIGN_BASELINE,
143            #[cfg(feature = "v4_12")]
144            Self::BaselineFill | Self::Baseline => ffi::GTK_ALIGN_BASELINE_FILL,
145            #[cfg(feature = "v4_12")]
146            Self::BaselineCenter => ffi::GTK_ALIGN_BASELINE_CENTER,
147            Self::__Unknown(value) => value,
148        }
149    }
150}
