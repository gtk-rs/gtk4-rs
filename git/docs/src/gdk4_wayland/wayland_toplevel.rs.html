14impl WaylandToplevel {
15    /// Returns the Wayland `xdg_toplevel` of a toplevel,
16    /// if it has one.
17    ///
18    /// # Returns
19    ///
20    /// a Wayland `xdg_toplevel`
21    #[cfg(all(feature = "v4_20", feature = "wayland_crate"))]
22    #[cfg_attr(docsrs, doc(cfg(all(feature = "v4_20", feature = "wayland_crate"))))]
23    #[doc(alias = "gdk_wayland_toplevel_get_xdg_toplevel")]
24    #[doc(alias = "get_xdg_toplevel")]
25    pub fn xdg_toplevel(&self) -> Option<XdgToplevel> {
26        use gdk::prelude::*;
27        let display = self.display().downcast::<crate::WaylandDisplay>().unwrap();
28        unsafe {
29            let toplevel_ptr = ffi::gdk_wayland_toplevel_get_xdg_toplevel(self.to_glib_none().0);
30            if toplevel_ptr.is_null() {
31                None
32            } else {
33                let cnx = display.connection();
34                let id =
35                    ObjectId::from_ptr(XdgToplevel::interface(), toplevel_ptr as *mut _).unwrap();
36
37                XdgToplevel::from_id(&cnx, id).ok()
38            }
39        }
40    }

42    /// Asynchronously obtains a handle for a surface that can be passed
43    /// to other processes.
44    ///
45    /// When the handle has been obtained, @callback will be called.
46    ///
47    /// It is an error to call this function on a surface that is already
48    /// exported.
49    ///
50    /// When the handle is no longer needed, [`unexport_handle()`][Self::unexport_handle()]
51    /// should be called to clean up resources.
52    ///
53    /// The main purpose for obtaining a handle is to mark a surface
54    /// from another surface as transient for this one, see
55    /// [`set_transient_for_exported()`][Self::set_transient_for_exported()].
56    ///
57    /// Before 4.12, this API could not safely be used multiple times,
58    /// since there was no reference counting for handles. Starting with
59    /// 4.12, every call to this function obtains a new handle, and every
60    /// call to [`drop_exported_handle()`][Self::drop_exported_handle()] drops
61    /// just the handle that it is given.
62    ///
63    /// Note that this API depends on an unstable Wayland protocol,
64    /// and thus may require changes in the future.
65    /// ## `callback`
66    /// callback to call with the handle
67    ///
68    /// # Returns
69    ///
70    /// [`true`] if the handle has been requested, [`false`] if
71    ///   an error occurred.
72    #[doc(alias = "gdk_wayland_toplevel_export_handle")]
73    pub fn export_handle<P: Fn(&WaylandToplevel, Result<&str, glib::BoolError>) + 'static>(
74        &self,
75        callback: P,
76    ) -> bool {
77        let callback_data: Box_<P> = Box_::new(callback);
78        unsafe extern "C" fn callback_func<
79            P: Fn(&WaylandToplevel, Result<&str, glib::BoolError>) + 'static,
80        >(
81            toplevel: *mut ffi::GdkWaylandToplevel,
82            handle: *const libc::c_char,
83            user_data: glib::ffi::gpointer,
84        ) {
85            let toplevel = from_glib_borrow(toplevel);
86            let handle: Borrowed<Option<glib::GString>> = from_glib_borrow(handle);
87            let callback = &*(user_data as *mut P);
88            if let Some(handle) = handle.as_ref() {
89                (*callback)(&toplevel, Ok(handle.as_str()))
90            } else {
91                (*callback)(&toplevel, Err(glib::bool_error!("Failed to export a handle. The compositor probably doesn't implement the xdg-foreign protocol")))
92            }
93        }
94        let callback = Some(callback_func::<P> as _);
95        unsafe extern "C" fn destroy_func_func<
96            P: Fn(&WaylandToplevel, Result<&str, glib::BoolError>) + 'static,
97        >(
98            data: glib::ffi::gpointer,
99        ) {
100            let _callback = Box_::from_raw(data as *mut P);
101        }
102        let destroy_call3 = Some(destroy_func_func::<P> as _);
103        let super_callback0: Box_<P> = callback_data;
104        unsafe {
105            from_glib(ffi::gdk_wayland_toplevel_export_handle(
106                self.to_glib_none().0,
107                callback,
108                Box_::into_raw(super_callback0) as *mut _,
109                destroy_call3,
110            ))
111        }
112    }
113}
